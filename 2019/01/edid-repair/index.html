#!/usr/bin/python3
import sys
import os

with open(sys.argv[2], "rb") as f:
    edid = f.read()

for i, byt in enumerate(edid):
    os.system("echo Y | i2cset %d 0x50 0x%02x 0x%02x b" % (int(sys.argv[1]), i, byt))
    os.system("sleep 0.1")
EOS
